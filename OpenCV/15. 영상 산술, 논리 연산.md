## 엠보싱 효과 : 표면적으로 보이게

#include "opencv2/opencv.hpp"

#include <iostream>

using namespace cv;

using namespace std;

int main() {

	Mat src = imread("rose.bmp", IMREAD_GRAYSCALE);

	if (src.empty()) {
		cout << "Image load failed!" << endl;
		return 0;
	}

	float data[] = { -1,-1,0,-1,0,1,0,1,1 };
	Mat emboss(3, 3, CV_32FC1, data);

	Mat dst;
	filter2D(src, dst, -1, emboss, Point(-1, -1), 128);

	imshow("src", src);
	imshow("dst", dst);

	waitKey();
	destroyAllWindows();
}

## 블러링 : 부드럽게

### 평균값 필터 이용

- 마스크의 크기가 커질수록 부드러운 느낌의 결과 영상을 생성한다.

#include "opencv2/opencv.hpp"

#include <iostream>

using namespace cv;

using namespace std;

int main() {

	Mat src = imread("rose.bmp", IMREAD_GRAYSCALE);

	if (src.empty()) {
		cout << "Image load failed!" << endl;
		return 0;
	}

	imshow("src", src);

	Mat dst;
	for (int ksize = 3; ksize <= 7; ksize += 2) {
		blur(src, dst, Size(ksize, ksize));

		String desc = format("Mean: %dx%d", ksize, ksize);
		putText(dst, desc, Point(10, 30), FONT_HERSHEY_SIMPLEX, 1.0, Scalar(255), 1, LINE_AA);
		imshow("dst", dst);

		waitKey(0);

	}


	destroyAllWindows();
}

## 가우시안 : 평균값 필터보다 자연스럽다.

- 가우시안 분포 : 평균을 중심으로 좌우 대칭의 종 모양을 갖는 확률분포 = 정규 분포

- sigma : 가우시안 표준 편차


#include "opencv2/opencv.hpp"

#include <iostream>

using namespace cv;

using namespace std;

int main() {

	Mat src = imread("rose.bmp", IMREAD_GRAYSCALE);

	if (src.empty()) {
		cout << "Image load failed!" << endl;
		return 0;
	}

	imshow("src", src);

	Mat dst;
	for (int sigma = 1; sigma <= 5; sigma++) {
		GaussianBlur(src, dst, Size(), (double)sigma);

		String text = format("sigma = %d", sigma);
		putText(dst, text, Point(10, 30), FONT_HERSHEY_SIMPLEX, 1.0, Scalar(255), 1, LINE_AA);

		
		imshow("dst", dst);

		waitKey(0);

	}


	destroyAllWindows();
}

## 샤프닝 : 날카롭게 (객체 윤곽 뚜렷히)

- 영상 에지 근방에서 픽셀 값의 명암비가 커지도록

- 언샤프 마스크 필터 : 블러링이 적용된 영상을 이용하여 역으로 날카로운 영상을 만드는 필터

    g(x,y) = f(x,y) - f_(x,y)

    f_(x,y)는 블러링된 영상

    h(x,y) = f(x,y) + a*g(x,y) ==> 샤프닝이 적용된 결과 영상이 생성된다.(a는 날카로운 정도 조절)


#include "opencv2/opencv.hpp"

#include <iostream>

using namespace cv;

using namespace std;

int main() {

	Mat src = imread("rose.bmp", IMREAD_GRAYSCALE);

	if (src.empty()) {
		cout << "Image load failed!" << endl;
		return 0;
	}

	imshow("src", src);

	Mat dst;
	for (int sigma = 1; sigma <= 5; sigma++) {
		Mat blurred;
		GaussianBlur(src, blurred, Size(), (double)sigma);

		float alpha = 1.f;
		Mat dst = (1 + alpha) * src - alpha * blurred;

		String desc = format("sigma : %d", sigma);
		putText(dst, desc, Point(10, 30), FONT_HERSHEY_SIMPLEX, 1.0, Scalar(255), 1, LINE_AA);
		
		imshow("dst", dst);

		waitKey(0);

	}


	destroyAllWindows();
}

## 양방향 필터

대부분의 영상의 가우시안 잡음을 가우시안 필터를 사용하면 제거할 수 있는데 에지까지 적용되어 에지 성분도 함께 감소하는데 이런 경우를 보완!

#include "opencv2/opencv.hpp"

#include <iostream>

using namespace cv;

using namespace std;

int main() {

	Mat src = imread("rose.bmp", IMREAD_GRAYSCALE);

	if (src.empty()) {
		cout << "Image load failed!" << endl;
		return 0;
	}

	Mat noise(src.size(), CV_32SC1);
	randn(noise, 0, 10);
	add(src, noise, src, Mat(), CV_8U);

	Mat dst1;
	GaussianBlur(src, dst1, Size(), 5);

	Mat dst2;
	bilateralFilter(src, dst2, -1, 10, 5);
    //10은 시그마컬러(색공간 가우시안 표준편차)
    //5는 시그마스페이스(좌표공간 가우시안 표준편차)
    //둘 다 커질수록 흐려짐.

	imshow("src", src);
	imshow("dst1", dst1);
	imshow("dst2", dst2);
	waitKey();
	destroyAllWindows();
}
